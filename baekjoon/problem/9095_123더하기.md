```python
def dfs(target_num, add_num, value):
    global case
    value = value + add_num
    if value == target_num:
        case += 1
    if value > target_num:
        return
    dfs(target_num, 1, value)
    dfs(target_num, 2, value)
    dfs(target_num, 3, value)

result = []
n = int(input())
nums = [int(input()) for _ in range(n)]

for i in range(n):
    case = 0
    dfs(nums[i], 0, 0)
    result.append(case)

print("\n".join(map(str, result)))
```
