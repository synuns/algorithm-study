n = int(input())
table = [list(map(int, input().split())) for _ in range(n)]
result = 0

def dfs(deadline, days, value):
    global result
    if days > deadline:
        return
    elif result < value:
        result = value
    if days < deadline:
        dfs(deadline, days+1, value)
        dfs(deadline, days+table[days][0], value+table[days][1])

dfs(n, 0, 0)
print(result)
