    while i>0 and a[i-1]<=a[i]:i-=1

    j=n-1
    while a[j]>=a[i-1]:j-=1

    j=n-1
    while i<j:
        x=a[i];a[i]=a[j];a[j]=x
        i+=1;j-=1
    return a
