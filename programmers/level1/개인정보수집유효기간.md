function calExpiredPeriod(today, join){
    let [todayYear, todayMonth, todayDate] = today.split('.').map(el => Number(el));
    let [joinYear, joinMonth, joinDate] = join.split('.').map(el => Number(el));
    let period = (todayYear - joinYear) * 12 + (todayMonth - joinMonth);
    if(todayDate - joinDate < 0) period -= 1;
    return period;
}

function isExpired(period, expiration) {
    if(period > expiration) return false;
    return true;
}

function solution(today, terms, privacies) {
    const expired = [];
    terms = terms.reduce((acc, val) => {
        const [type, period] = val.split(' ');
        return {...acc, [type]: Number(period)};
    }, {});
    
    for(let i=0; i<privacies.length; i++) {
        const [joinDate, type] = privacies[i].split(' ');
        const expiration = calExpiredPeriod(today, joinDate);
        if(isExpired(terms[type], expiration)) expired.push(i+1);
    }
    
    return expired;
}
