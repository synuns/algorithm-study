```python
def solution(board, moves):
    answer = 0
    pocket = []
    for i in moves:
        for j in range(len(board)):
            if board[j][i-1] != 0:
                pocket.append(board[j][i-1])
                board[j][i-1] = 0
                break
        if len(pocket) >= 2 and pocket[-1] == pocket[-2]:
            pocket.pop(-1)
            pocket.pop(-1)
            answer += 2
    return answer
```
