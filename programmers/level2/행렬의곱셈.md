function solution(arr1, arr2) {
    const col = arr1.length;
    const row = arr2[0].length;
    const answer = [];
    for(let i = 0; i < col; i++){
        answer.push([]);
        for(let j = 0; j < row; j++){
            let element = 0;
            for(let k=0; k<arr1[i].length; k++){
                element += arr1[i][k] * arr2[k][j]
            }
            answer[i].push(element);
        }
    }
    return answer;
}
